Tangle.formats

 
Tangle.formats.metric = function (value) {
   var division = 1, modifier = "";
   if (value >= 1e12) {
      division = 1e12;
      modifier = "T";
   }
   else if (value >= 1e9) {
      division = 1e9;
      modifier = "G";
   }
   else if (value >= 1e6) {
      division = 1e6;
      modifier = "M";
   }
   else if (value >= 1e3) {
      division = 1e3;
      modifier = "k";
   }

   return sprintf("%.2f", value/division) + " " + modifier;
};
